공유 설정

아래 주소를 등록하면 모든 기기가 같은 트리를 함께 보게 돼요. 비워두면 이 기기에만 저장돼요.

🔧 아직 주소가 없다면? 5분만에 무료로 만드는 방법
  1. 구글 스프레드시트를 새로 만들고, 1행에 id, message, name, ts 를 순서대로 적습니다.
  2. 상단 메뉴 확장 프로그램 → Apps Script를 클릭합니다.
  3. 기본 코드를 모두 지우고 아래 코드를 붙여넣습니다.
  4. 저장 후 배포 → 새 배포를 클릭, 유형은 웹 앱을 선택합니다.
  5. 실행 계정 나, 액세스 권한 모든 사용자로 설정 후 배포합니다.
  6. 발급된 URL을 복사해 위 입력창에 붙여넣고 저장하세요.
function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var rows = sheet.getDataRange().getValues(); var out = []; for (var i = 1; i < rows.length; i++) { if (!rows[i][0]) continue; out.push({ id: rows[i][0], message: rows[i][1], name: rows[i][2], ts: rows[i][3] }); } return ContentService.createTextOutput(JSON.stringify(out)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var body = JSON.parse(e.postData.contents); var id = Utilities.getUuid(); var ts = new Date().toISOString(); sheet.appendRow([ id, String(body.message || '').substring(0, 200), String(body.name || '').substring(0, 30), ts ]); return ContentService.createTextOutput(JSON.stringify({ ok: true, id: id })) .setMimeType(ContentService.MimeType.JSON); }
